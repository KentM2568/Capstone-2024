 while (true) {
        cout <<"Do you wish to find out the total price of an item after tax (AT) OR The original price of an item before tax (BT): ";
        cin >> strUserInput;

        //First if statement, based on the inital user input, calculates the user input after taxes
        if (strUserInput == "AT") {
            
            //Getting number inputs from the user
            cout << "Please enter a number the price of the item before tax: ";
            cin >> numUserInput;
            
            cout << "Thank you, now please enter the provincial tax (without the % symbol): ";
            cin >> provincialTax;
            
            cout << "Thank you, now please enter the federal tax (without the % symbol): ";
            cin >> federalTax;
            
            //Tax calculations (after taxes)
            totalTax = provincialTax + federalTax;
            provincialPrice = numUserInput * (provincialTax/100 + 1);
            federalPrice = numUserInput * (federalTax/100 + 1);
            totalPrice = numUserInput * (totalTax/100 + 1);
            
            //Output of the final calculations to the user with set precision
            cout << fixed;
            cout << "The price with provincial tax is $" << setprecision(2) << provincialPrice << "\n";
            cout << "The price with federal tax is $" << setprecision(2) << federalPrice << "\n";
            cout << "The price with total tax is $" << setprecision(2) << totalPrice << "\n\n\n";
            
            //Ends the while loop, completing the program
            break;
        }
        
        //Second if statement, also based on the initial user input, calculates the user's input before taxes
        else if (strUserInput == "BT"){
            
            //Getting number inputs from the user
            cout << "Please enter a number the price of the item after tax: ";
            cin >> numUserInput;
            
            cout << "Thank you, now please enter the provincial tax(without the % symbol): ";
            cin >> provincialTax;
            
            cout << "Thank you, now please enter the federal tax (without the % symbol): ";
            cin >> federalTax;
            
            //Tax calculations (before taxes)
            totalTax = provincialTax + federalTax;
            provincialPrice = numUserInput / (provincialTax/100 + 1);
            federalPrice = numUserInput / (federalTax/100 + 1);
            totalPrice = numUserInput / (totalTax/100 + 1);
            
            //Output of the final calculations to the user with set precision
            cout << fixed;
            cout << "The price minus the  provincial tax is $" << setprecision(2) << provincialPrice << "\n";
            cout << "The price minus federal tax is $" << setprecision(2) << federalPrice << "\n";
            cout << "The original price is $" << setprecision(2) << totalPrice << "\n\n\n";
        
            //Ends the while loop, completing the program
            break;
        }
        //Else statement just in case the user enters an invalid input on the *initial* input
        else {
           //tells the user their input was invalid and sends them to the begining of the loop
           cout << "\nInvalid Input Please Try Again!\n\n";
        }
        
    }

      int evaluatePosition(string playerOneSymbol, string playerTwoSymbol)
      {
          //If the game is over, that means that someone has won, meaning that the score needs to be very large
          if (checkGameOver())
          {
              if (checkVertWin(playerOneSymbol, playerTwoSymbol) == 1)
              {
                  return 2000;
              }
              else if (checkVertWin(playerOneSymbol, playerTwoSymbol) == 2)
              {
                  return -2000;
              }
              if (checkHorizontWin(playerOneSymbol, playerTwoSymbol) == 1)
              {
                  return 2000;
              }
              else if (checkHorizontWin(playerOneSymbol, playerTwoSymbol) == 2)
              {
                  return -2000;
              }
              if (checkDiagonalWin(playerOneSymbol, playerTwoSymbol) == 1)
              {
                  return 2000;
              }
              else if (checkDiagonalWin(playerOneSymbol, playerTwoSymbol) == 2)
              {
                  return -2000;
              }
              return 0;
          }
  
          //These variables are used to store the evalutaion of each direction, and then return it as a score variable
          int score = 0;
          int numEmpty = getNumEmptySpaces();
          int totalSpaces = 42;
          double emptyRatio = static_cast<double>(numEmpty) / totalSpaces;
          double vertWeight = 1.0;
          double horizWeight = 1.0;
          double diagWeight = 1.0;
  
          //In the early game, horizontal wins have a higher value, as they are usually pretty easy to build in the early game
          if (emptyRatio >= 0.7)
          {
              vertWeight = 1;
              horizWeight = 1.5;
              diagWeight = 1.25;
  
          }
          //In the mid game, everything is almost equal, except that vertical scores are worth more since the game needs to get higher and higher in the board
          else if (emptyRatio >= 0.4)
          {
              vertWeight = 1.25;
              horizWeight = 1;
              diagWeight = 1;
          }
          //At the end game everything is more a bit more, since winning is very important when there are limited spots
          else
          {
              vertWeight = 1.25;
              horizWeight = 1.25;
              diagWeight = 1.25;
          }
  
  
          //adds these weighted score values to a score variable and returns it
          score += (vertWeight * evaluateVertical(playerOneSymbol, playerTwoSymbol));
          score += (horizWeight * evaluateHorizontal(playerOneSymbol, playerTwoSymbol));
          score += (diagWeight * evaluateDiagonal(playerOneSymbol, playerTwoSymbol));
  
          return score;
  
      }

The C++ final project was very open ended, and I had a lot of freedom to chose what to do. One thing that I had no previous experience with was working with artificial intelligence, and I always wanted to delve into that subject, but this project presented the perfect opportunity to do so. So I knew that I wanted to make an AI, and I had to decide on what it would do, and I laneded on the game of connect 4, which would be relatively easy to make an implementation for, to allow me enough time to learn and create an AI for it. For the first week of the month that we had to do this project, I didn't really do much learning, it was mostly just recalling my previous programming knowledge, to make a working connect 4 game using 2D arrays. But when the second week started, I delved into independed research about algorithms and so many things that I had no experience with. After many hours of trial, error, and learning by the 4th week I had a working AI, which was comprised of 2 main elements. The first one is the functions used to statically evalue the value of a given position, this function is vital when determining the best move, as the AI needs to know what moves are good and what moves to are bad. The second and most important part of the AI is the minimax algorithm, the algoirthm used to go through each possible move, and determine the one that has the greates value for the given player. This algorithm goes to a certain depth, meaning moves into the future, and then returns to give a point value to each of the 7 possible moves in the average connect 4 board. This minimax algoritim is then finally used in the findBestMove functions, to actually return the best possible move in a given position, for a given player. All of these elements combined, make it so the computer has strategic response, and "thinks" about what the best move is, making it an AI. This project was by far the biggest learning experience for me in this entire course, and I really enjoyed the challenge that it brought.
